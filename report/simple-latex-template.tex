\begin{algo}[H]
  \caption{Frank-Wolfe Algorithm}
  \label{alg:FW}
\SetAlgoNoLine\DontPrintSemicolon
  Let $\x^{(0)} \in \domain$\;
  \For{$t=0\dots T$}{
  %
  Compute $\s := \displaystyle\argmin_{\s\in \domain}\textstyle\left\langle \s, \nabla f(\x^{(t)}) \right\rangle$\!\!\!\vspace{-1mm}\;
  Let $\stepsize := \frac2{t+2}$\;
  Update $\x^{(t+1)}:= (1-\stepsize)\x^{(t)}+\stepsize\s$\;
  }%FOR
\end{algo}
